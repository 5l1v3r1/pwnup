Introduction

pwnup is a utility to automate pwntools

Install

pip install pwnup

from pwn import *

r = process('bash')

def main():
  r.send('ls -la\n')
  print(r.recvuntil(' ubuntu   17 Jun  8 01:23 shell\n'))
  r.send('whoami\n')
  print(r.recvuntil('ubuntu\n'))
  r.send('cat secrets\n')
